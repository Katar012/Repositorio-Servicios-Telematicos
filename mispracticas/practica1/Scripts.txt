Vagrant.configure("2") do |config|
  config.vm.box = "bento/ubuntu-22.04"

  common_packages = <<-SHELL
    apt-get update
    DEBIAN_FRONTEND=noninteractive apt-get install -y \
      net-tools curl wget dnsutils iproute2 tcpdump ca-certificates
  SHELL

  config.vm.define "servidor1" do |srv1|
    srv1.vm.hostname = "servidor1"

    # Red "publica" del laboratorio: cliente <-> servidor1
    srv1.vm.network "private_network", ip: "192.168.60.3"

    # Red interna backend: servidor1 <-> servidor2
    srv1.vm.network "private_network", ip: "192.168.50.3", virtualbox__intnet: "backendnet"

    srv1.vm.provider "virtualbox" do |vb|
      vb.name = "parcial-servidor1"
      vb.memory = 1024
      vb.cpus = 1
    end

    srv1.vm.provision "shell", inline: common_packages
    srv1.vm.provision "shell", inline: "apt-get install -y ufw openssh-server"
  end

  config.vm.define "servidor2" do |srv2|
    srv2.vm.hostname = "servidor2"

    # Solo backend, no debe quedar expuesto directamente al cliente
    srv2.vm.network "private_network", ip: "192.168.50.2", virtualbox__intnet: "backendnet"

    srv2.vm.provider "virtualbox" do |vb|
      vb.name = "parcial-servidor2"
      vb.memory = 1024
      vb.cpus = 1
    end

    srv2.vm.provision "shell", inline: common_packages
    srv2.vm.provision "shell", inline: "apt-get install -y vsftpd openssl"
  end

  config.vm.define "cliente" do |cli|
    cli.vm.hostname = "cliente"

    cli.vm.network "private_network", ip: "192.168.60.2"

    cli.vm.provider "virtualbox" do |vb|
      vb.name = "parcial-cliente"
      vb.memory = 1024
      vb.cpus = 1
    end

    cli.vm.provision "shell", inline: common_packages
    cli.vm.provision "shell", inline: "apt-get install -y openssl lftp"
  end
end
